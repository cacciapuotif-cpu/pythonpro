```python
# ========================================
# AVVISI
# ========================================

def get_avvisi(
    db: Session,
    ente_erogatore: Optional[str] = None,
    active_only: bool = True,
    skip: int = 0,
    limit: int = 200,
) -> list:
    query = db.query(models.Avviso)
    if active_only:
        query = query.filter(models.Avviso.is_active == True)
    if ente_erogatore:
        query = query.filter(models.Avviso.ente_erogatore == ente_erogatore)
    return query.order_by(models.Avviso.ente_erogatore, models.Avviso.codice).offset(skip).limit(limit).all()


def get_avviso(db: Session, avviso_id: int):
    return db.query(models.Avviso).filter(models.Avviso.id == avviso_id).first()


def create_avviso(db: Session, avviso: schemas.AvvisoCreate):
    existing = db.query(models.Avviso).filter(
        models.Avviso.codice == avviso.codice.strip(),
        models.Avviso.ente_erogatore == avviso.ente_erogatore.strip(),
    ).first()
    if existing:
        raise ValueError(f"Avviso '{avviso.codice}' per '{avviso.ente_erogatore}' già esistente")
    # Se il template è specificato, esegui pointer swap (un template → un solo avviso)
    if avviso.template_id is not None:
        db.query(models.Avviso).filter(
            models.Avviso.template_id == avviso.template_id
        ).update({"template_id": None})
    db_obj = models.Avviso(
        codice=avviso.codice.strip(),
        ente_erogatore=avviso.ente_erogatore.strip(),
        descrizione=avviso.descrizione,
        template_id=avviso.template_id,
        is_active=avviso.is_active,
    )
    db.add(db_obj)
    db.commit()
    db.refresh(db_obj)
    return db_obj


def update_avviso(db: Session, avviso_id: int, avviso: schemas.AvvisoUpdate):
    db_obj = get_avviso(db, avviso_id)
    if not db_obj:
        return None
    update_data = avviso.model_dump(exclude_unset=True)
    # Pointer swap: se stiamo collegando un template, rimuovilo dagli altri avvisi
    if "template_id" in update_data and update_data["template_id"] is not None:
        db.query(models.Avviso).filter(
            models.Avviso.template_id == update_data["template_id"],
            models.Avviso.id != avviso_id,
        ).update({"template_id": None})
    for key, value in update_data.items():
        setattr(db_obj, key, value)
    db.commit()
    db.refresh(db_obj)
    return db_obj


def delete_avviso(db: Session, avviso_id: int):
    db_obj = get_avviso(db, avviso_id)
    if db_obj:
        db_obj.is_active = False
        db.commit()
        db.refresh(db_obj)
    return db_obj
```

```python
def _resolve_financial_template(
    db: Session,
    *,
    progetto: models.Project,
    ente_erogatore: str,
    avviso_id: Optional[int],
    template_id: Optional[int] = None,
):
    normalized_ente = _normalize_optional_text(ente_erogatore)
    project_ente = _normalize_optional_text(getattr(progetto, "ente_erogatore", None))

    if template_id is not None:
        template = db.query(models.ContractTemplate).filter(
            models.ContractTemplate.id == template_id,
            models.ContractTemplate.is_active == True,
        ).first()
        if not template:
            raise ValueError("Template piano finanziario non trovato")
        if template.ambito_template != "piano_finanziario":
            raise ValueError("Il template selezionato non è un template piano finanziario")
        if project_ente and template.ente_erogatore and template.ente_erogatore != project_ente:
            raise ValueError("Il template piano finanziario selezionato non coincide con l'ente erogatore del progetto")
        return template

    # Risolvi tramite avviso: l'avviso ha già template_id
    if avviso_id is not None:
        avviso = db.query(models.Avviso).filter(
            models.Avviso.id == avviso_id,
            models.Avviso.is_active == True,
            models.Avviso.template_id.isnot(None),
        ).first()
        if avviso and avviso.template_id:
            return db.query(models.ContractTemplate).filter(
                models.ContractTemplate.id == avviso.template_id,
                models.ContractTemplate.is_active == True,
            ).first()

    # Fallback: primo avviso attivo dell'ente con template collegato
    if normalized_ente:
        avviso = db.query(models.Avviso).filter(
            models.Avviso.ente_erogatore == normalized_ente,
            models.Avviso.is_active == True,
            models.Avviso.template_id.isnot(None),
        ).order_by(models.Avviso.id.desc()).first()
        if avviso:
            return db.query(models.ContractTemplate).filter(
                models.ContractTemplate.id == avviso.template_id,
                models.ContractTemplate.is_active == True,
            ).first()

    return None
```

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from typing import List, Optional
from database import get_db
import crud
import schemas

router = APIRouter(prefix="/api/v1/avvisi", tags=["avvisi"])


@router.get("/", response_model=List[schemas.AvvisoResponse])
def list_avvisi(
    ente_erogatore: Optional[str] = None,
    active_only: bool = True,
    skip: int = 0,
    limit: int = 200,
    db: Session = Depends(get_db),
):
    return crud.get_avvisi(db, ente_erogatore=ente_erogatore, active_only=active_only, skip=skip, limit=limit)


@router.post("/", response_model=schemas.AvvisoResponse, status_code=201)
def create_avviso(avviso: schemas.AvvisoCreate, db: Session = Depends(get_db)):
    try:
        return crud.create_avviso(db, avviso)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))


@router.get("/{avviso_id}", response_model=schemas.AvvisoResponse)
def get_avviso(avviso_id: int, db: Session = Depends(get_db)):
    obj = crud.get_avviso(db, avviso_id)
    if not obj:
        raise HTTPException(status_code=404, detail="Avviso non trovato")
    return obj


@router.put("/{avviso_id}", response_model=schemas.AvvisoResponse)
def update_avviso(avviso_id: int, avviso: schemas.AvvisoUpdate, db: Session = Depends(get_db)):
    try:
        obj = crud.update_avviso(db, avviso_id, avviso)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    if not obj:
        raise HTTPException(status_code=404, detail="Avviso non trovato")
    return obj


@router.delete("/{avviso_id}", response_model=schemas.AvvisoResponse)
def delete_avviso(avviso_id: int, db: Session = Depends(get_db)):
    obj = crud.delete_avviso(db, avviso_id)
    if not obj:
        raise HTTPException(status_code=404, detail="Avviso non trovato")
    return obj
```

```javascript
import React, { useEffect, useState } from 'react';
import { getAvvisi, createAvviso, updateAvviso, deleteAvviso, getContractTemplates } from '../services/apiService';
import './AvvisiManager.css';

const EMPTY_FORM = { codice: '', ente_erogatore: '', descrizione: '', template_id: '', is_active: true };

export default function AvvisiManager() {
  const [avvisi, setAvvisi] = useState([]);
  const [templates, setTemplates] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const [showModal, setShowModal] = useState(false);
  const [editing, setEditing] = useState(null);
  const [form, setForm] = useState(EMPTY_FORM);
  const [saving, setSaving] = useState(false);
  const [formError, setFormError] = useState('');

  const load = async () => {
    setLoading(true);
    setError('');
    try {
      const [av, tmpl] = await Promise.all([
        getAvvisi({ active_only: false }),
        getContractTemplates({ ambito_template: 'piano_finanziario', limit: 100 }),
      ]);
      setAvvisi(Array.isArray(av) ? av : []);
      setTemplates(Array.isArray(tmpl) ? tmpl : (tmpl?.items || []));
    } catch (e) {
      setError('Errore nel caricamento avvisi.');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => { load(); }, []);

  const openCreate = () => {
    setEditing(null);
    setForm(EMPTY_FORM);
    setFormError('');
    setShowModal(true);
  };

  const openEdit = (av) => {
    setEditing(av);
    setForm({
      codice: av.codice || '',
      ente_erogatore: av.ente_erogatore || '',
      descrizione: av.descrizione || '',
      template_id: av.template_id ? String(av.template_id) : '',
      is_active: av.is_active !== false,
    });
    setFormError('');
    setShowModal(true);
  };

  const handleSave = async () => {
    if (!form.codice.trim()) { setFormError('Codice obbligatorio'); return; }
    if (!form.ente_erogatore.trim()) { setFormError('Ente erogatore obbligatorio'); return; }
    setSaving(true);
    setFormError('');
    try {
      const payload = {
        codice: form.codice.trim(),
        ente_erogatore: form.ente_erogatore.trim(),
        descrizione: form.descrizione.trim() || null,
        template_id: form.template_id ? parseInt(form.template_id) : null,
        is_active: form.is_active,
      };
      if (editing) {
        await updateAvviso(editing.id, payload);
      } else {
        await createAvviso(payload);
      }
      setShowModal(false);
      await load();
    } catch (e) {
      setFormError(e?.response?.data?.detail || 'Errore nel salvataggio');
    } finally {
      setSaving(false);
    }
  };

  const handleDelete = async (av) => {
    if (!window.confirm(`Disattivare avviso "${av.codice}"?`)) return;
    try {
      await deleteAvviso(av.id);
      await load();
    } catch (e) {
      setError(e?.response?.data?.detail || 'Errore nella disattivazione');
    }
  };

  const templateName = (id) => {
    const t = templates.find(t => t.id === id);
    return t ? t.nome_template : null;
  };

  return (
    <div className="avvisi-manager">
      <div className="page-header">
        <div>
          <span className="page-eyebrow">Configurazione</span>
          <h2>Avvisi di Finanziamento</h2>
          <p>Catalogo avvisi collegati ai template piani finanziari.</p>
        </div>
        <div className="page-header-actions">
          <button className="btn-primary" onClick={openCreate}>+ Nuovo Avviso</button>
        </div>
      </div>

      {error && <div className="banner error">{error}</div>}
      {loading && <div className="loading-state">Caricamento...</div>}

      {!loading && (
        <div className="avvisi-table-wrap">
          <table className="avvisi-table">
            <thead>
              <tr>
                <th>Codice</th>
                <th>Ente Erogatore</th>
                <th>Descrizione</th>
                <th>Template Piano</th>
                <th>Stato</th>
                <th></th>
              </tr>
            </thead>
            <tbody>
              {avvisi.length === 0 && (
                <tr><td colSpan={6} style={{textAlign:'center',color:'#999',padding:'24px'}}>Nessun avviso. Clicca "+ Nuovo Avviso" per iniziare.</td></tr>
              )}
              {avvisi.map(av => (
                <tr key={av.id} className={!av.is_active ? 'avviso-inactive' : ''}>
                  <td><strong>{av.codice}</strong></td>
                  <td><span className="badge-ente">{av.ente_erogatore}</span></td>
                  <td>{av.descrizione || <span style={{color:'#aaa'}}>—</span>}</td>
                  <td>
                    {av.template_id
                      ? <span className="badge-template">📋 {templateName(av.template_id) || `Template #${av.template_id}`}</span>
                      : <span className="badge-no-template">— nessuno</span>}
                  </td>
                  <td>{av.is_active ? '✅ Attivo' : '⏸ Inattivo'}</td>
                  <td style={{whiteSpace:'nowrap'}}>
                    <button className="btn-sm" onClick={() => openEdit(av)}>Modifica</button>
                    {av.is_active && (
                      <button className="btn-sm btn-danger" style={{marginLeft:6}} onClick={() => handleDelete(av)}>Disattiva</button>
                    )}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}

      {showModal && (
        <div className="avvisi-modal-overlay" onClick={e => { if(e.target===e.currentTarget) setShowModal(false); }}>
          <div className="avvisi-modal">
            <h3>{editing ? 'Modifica Avviso' : 'Nuovo Avviso'}</h3>

            {formError && <div className="banner error" style={{marginBottom:12}}>{formError}</div>}

            <div className="form-group">
              <label>Codice avviso *</label>
              <input value={form.codice} onChange={e => setForm(f=>({...f,codice:e.target.value}))} placeholder="Es. 9/2026" />
            </div>
            <div className="form-group">
              <label>Ente Erogatore *</label>
              <input value={form.ente_erogatore} onChange={e => setForm(f=>({...f,ente_erogatore:e.target.value}))} placeholder="Es. FORMAZIENDA" />
            </div>
            <div className="form-group">
              <label>Descrizione</label>
              <input value={form.descrizione} onChange={e => setForm(f=>({...f,descrizione:e.target.value}))} placeholder="Descrizione breve (opzionale)" />
            </div>
            <div className="form-group">
              <label>Template Piano Finanziario collegato</label>
              <select value={form.template_id} onChange={e => setForm(f=>({...f,template_id:e.target.value}))}>
                <option value="">— nessuno —</option>
                {templates.map(t => (
                  <option key={t.id} value={t.id}>{t.nome_template}{t.ente_erogatore ? ` · ${t.ente_erogatore}` : ''}</option>
                ))}
              </select>
              <small style={{color:'#666',fontSize:'12px'}}>Se collegato, il piano di questo avviso userà automaticamente questo template.</small>
            </div>
            <div className="form-group">
              <label>
                <input type="checkbox" checked={form.is_active} onChange={e => setForm(f=>({...f,is_active:e.target.checked}))} style={{marginRight:6}} />
                Avviso attivo
              </label>
            </div>

            <div className="modal-actions">
              <button className="btn-secondary" onClick={() => setShowModal(false)} disabled={saving}>Annulla</button>
              <button className="btn-primary" onClick={handleSave} disabled={saving}>{saving ? 'Salvataggio...' : 'Salva'}</button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
```
